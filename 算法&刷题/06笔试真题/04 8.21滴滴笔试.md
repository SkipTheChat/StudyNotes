package wpt;


import java.util.*;
import java.util.List;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        if(n == 1){
            System.out.println(1);
            return;
        }
        long[] arr = new long[n * n];
        long[][] a = new long[n][n];
        arr[0] = 1;
        arr[1] = 1;
        for (int i = 2; i < n * n; i++) {
            arr[i] = cal(i);
        }

        int top = 0;
        int left = 0;
        int bottom = n - 1;
        int right = n - 1;
        int index = n * n - 1;
        while(left < right && top < bottom){
            for (int i = left; i < right; i++) {
                a[top][i] = arr[index--];
            }

            for (int i = top; i < bottom; i++) {
                a[i][right] = arr[index--];
            }

            for (int i = right; i > left; i--) {
                a[bottom][i] = arr[index--];
            }

            for (int i = bottom; i > top ; i--) {
                a[i][left] = arr[index--];
            }
            top++;
            bottom--;
            left++;
            right--;
        }
        if(n % 2 == 1){
            a[n / 2][n / 2] = 1;
        }

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                System.out.print(a[i][j] + " ");
            }
            System.out.println();
        }

    }

    private static long cal(int i) {
        long a = 1;
        long b = 1;
        long sum = 0;
        for (int j = 1; j < i; j++) {
            sum = a + b;
            a = b;
            b = sum;

        }
        return sum;
    }
}
