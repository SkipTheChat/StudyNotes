# [字符串相乘](https://leetcode-cn.com/problems/multiply-strings/)

- 题目描述：

```java
给定两个以字符串形式表示的非负整数 num1 和 num2，返回 num1 和 num2 的乘积，它们的乘积也表示为字符串形式。

示例 1:

输入: num1 = "2", num2 = "3"
输出: "6"

示例 2:

输入: num1 = "123", num2 = "456"
输出: "56088"

说明：
    num1 和 num2 的长度小于110。
    num1 和 num2 只包含数字 0-9。
    num1 和 num2 均不以零开头，除非是数字 0 本身。
    不能使用任何标准库的大数类型（比如 BigInteger）或直接将输入转换为整数来处理。
```

class Solution {
  
    public String multiply(String num1, String num2) {
        if("0".equals(num1) || "0".equals(num2)){
            return "0";
        }
        int index = 0;
        String sum = "";

        for (int i = num1.length() - 1; i >= 0; i--) {
            //从最后一位开始乘
            String str = multiple(Integer.parseInt(String.valueOf(num1.charAt(i))),num2,index);
            index++; //index控制相乘的时候后面加0
            sum = add(str,sum);
        }

        return sum;

    }


    //对两个字符串进行相加
    private String add(String str, String sum) {
        if("".equals(sum)){
            return str;
        }

        if("".equals(str)){
            return sum;
        }

        int i = str.length() - 1;
        int j = sum.length() - 1;
        StringBuilder sb = new StringBuilder();

        int carry = 0;
        while(i >= 0 || j >= 0){
            int a = 0;
            int b = 0;
            if(i >= 0){
                a = str.charAt(i) - '0';
            }
            if(j >= 0){
                b = sum.charAt(j) - '0';
            }

            int x = (a + b + carry) % 10;
            carry = (a + b + carry) / 10;
            sb.insert(0,x);
            i--;j--;
        }

        if(carry != 0){
            sb.insert(0,carry);
        }

        return sb.toString();

    }


	//对两个字符串进行相乘
    private String multiple(int a, String num2, int index) {
        StringBuilder sb = new StringBuilder();
        int carry = 0;
        for (int i = num2.length() - 1; i >= 0; i--) {
            int c = Integer.parseInt(String.valueOf(num2.charAt(i)));
            int x = (c * a + carry) % 10;
            carry = (c * a + carry) / 10;
            sb.insert(0,x);

        }

        if(carry != 0){
            sb.insert(0,carry);
        }

        while(index > 0){
            sb.append(0);
            index--;
        }
        return sb.toString();
    }

}

class Solution {
    public String multiply(String num1, String num2) {
        if (num1.equals("0") || num2.equals("0")) {
            return "0";
        }
        int[] res = new int[num1.length() + num2.length()];
        for (int i = num1.length() - 1; i >= 0; i--) {
            int n1 = num1.charAt(i) - '0';
            for (int j = num2.length() - 1; j >= 0; j--) {
                int n2 = num2.charAt(j) - '0';
                int sum = (res[i + j + 1] + n1 * n2);
                res[i + j + 1] = sum % 10;
                res[i + j] += sum / 10;
            }
        }

        StringBuilder result = new StringBuilder();
        for (int i = 0; i < res.length; i++) {
            if (i == 0 && res[i] == 0) continue;
            result.append(res[i]);
        }
        return result.toString();
    }
}
